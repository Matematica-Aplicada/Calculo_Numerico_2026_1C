\section*{Ejercicios de métodos de un paso}

\begin{exercise}[Método del punto medio sobre una EDO lineal]
Considere el problema
\[
  y'(t)=-y(t),\qquad y(0)=1,
\]
y el método del punto medio
\[
  y_{n+1}=y_n+h\,f\!\left(t_n+\frac{h}{2},\,y_n+\frac{h}{2}f(t_n,y_n)\right).
\]
\begin{enumerate}[(a)]
\item Escriba la función de transición $\Phi(t,y,h)$ del método.
\item Calcule el error de truncamiento local
\[
  \tau_{n+1}(h)=\frac{y(t_{n+1})-y(t_n)}{h}-\Phi(t_n,y(t_n),h)
\]
y determine su orden.
\item Halle la constante de Lipschitz de $\Phi$ respecto de la variable $y$ y deduzca una cota uniforme para $0<h\le 1$.
\item Use el teorema de convergencia de métodos de un paso del apunte para hallar un paso $h$ que garantice
\[
  \max_{t_n\le 10}|y(t_n)-y_n|<10^{-3}.
\]
\end{enumerate}
\end{exercise}

\begin{exercise}[Euler implícito sobre una EDO lineal]
Considere nuevamente
\[
  y'(t)=-y(t),\qquad y(0)=1,
\]
y el método de Euler implícito
\[
  y_{n+1}=y_n-hy_{n+1}.
\]
\begin{enumerate}[(a)]
\item Escriba la función de transición $\Phi(t,y,h)$ del método.
\item Calcule el error de truncamiento local y determine su orden.
\item Calcule la constante de Lipschitz de $\Phi$ respecto de $y$.
\item Use el teorema de convergencia de métodos de un paso para hallar un paso $h$ que garantice
\[
  \max_{t_n\le 10}|y(t_n)-y_n|<10^{-3}.
\]
\item Compare el paso obtenido aquí con el del ejercicio anterior.
\end{enumerate}
\end{exercise}

\begin{exercise}[Método de Heun]
Considere la EDO
\[
  y'(t)=-y(t)+t,\qquad y(0)=1,
\]
y el método de Heun
\begin{align*}
  k_1&=f(t_n,y_n),\\
  k_2&=f(t_n+h,\,y_n+hk_1),\\
  y_{n+1}&=y_n+\frac{h}{2}(k_1+k_2).
\end{align*}
\begin{enumerate}[(a)]
\item Escriba la función de transición $\Phi(t,y,h)$.
\item Halle el orden del error de truncamiento local.
\item Calcule una constante de Lipschitz de $\Phi$ en la variable $y$ para $0<h\le 1$.
\item Use el teorema del apunte para obtener un paso $h$ que garantice error global menor que $10^{-3}$ en el intervalo $[0,10]$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Problema rígido: Euler explícito e implícito]
Considere el problema
\[
  y'=-20\bigl(y-\cos t\bigr)-\sin t,\qquad y(0)=1.
\]
\begin{enumerate}[(a)]
\item Verifique que la solución exacta es $y(t)=\cos t$.
\item Escriba las funciones de transición $\Phi_E$ y $\Phi_I$ de Euler explícito e implícito.
\item Calcule las constantes de Lipschitz de $\Phi_E$ y $\Phi_I$ respecto de $y$.
\item Explique, usando la cota del teorema de convergencia de un paso, por qué el método explícito se deteriora cuando la constante de Lipschitz es grande.
\item Compare este análisis con el comportamiento cualitativo de ambos métodos sobre la ecuación test $y'=\lambda y$ con $\lambda<0$.
\end{enumerate}
\end{exercise}


\section*{Problemas de valores de contorno}

\begin{exercise}[Transporte-reacción periódico]
Considere el problema periódico
\[
  b\,u'(x)+\sigma u(x)=f(x),\qquad x\in(0,1),\qquad u(0)=u(1),\quad u'(0)=u'(1),
\]
con $b\in\R$ y $\sigma>0$. En la malla uniforme $x_j=jh$, $h=1/N$, use diferencia centrada para $u'$:
\[
  b\,\frac{u_{j+1}-u_{j-1}}{2h}+\sigma u_j=f(x_j),\qquad j=0,\dots,N-1,
\]
con índices módulo $N$.
\begin{enumerate}[(a)]
\item Verifique que los modos de Fourier discretos $w_j^{(k)}=e^{2\pi i kj/N}$ satisfacen automáticamente las condiciones de borde periódicas.
\item Muestre, por sustitución directa de $w^{(k)}$ en las ecuaciones del sistema discreto, que estos modos son autofunciones del operador de diferencias finitas y que los autovalores son
\[
  \lambda_k=\sigma+\frac{ib}{h}\sin\!\left(\frac{2\pi k}{N}\right).
\]
\item Deduzca que $|\lambda_k|\ge \sigma$ para todo $k$ y concluya, por el criterio de Fourier, que $\|(A_N)^{-1}\|_2\le 1/\sigma$.
\item Interprete este resultado como estabilidad en norma 2 del esquema discreto.
\end{enumerate}
\emph{Remark.} Si se escribe la matriz del sistema lineal, resulta ser circulante. Esto explica por qué los modos de Fourier discretos diagonalizan el operador, pero no es necesario usar esta observación para resolver el ejercicio.
\end{exercise}

\begin{exercise}[Advección-difusión-reacción periódica]
Considere
\[
  -\nu u''(x)+b\,u'(x)+\sigma u(x)=f(x),\qquad x\in(0,1),
\]
con condiciones periódicas, donde $\nu>0$, $b\in\R$ y $\sigma>0$.
\begin{enumerate}[(a)]
\item Discretice $u''$ y $u'$ con diferencias centradas y escriba las ecuaciones del esquema con índices módulo $N$.
\item Verifique que los modos de Fourier discretos satisfacen las condiciones de borde periódicas y úselo para diagonalizar el operador discreto.
\item Usando Fourier discreto, pruebe que los autovalores son
\[
  \lambda_k=\sigma+\frac{4\nu}{h^2}\sin^2\!\left(\frac{\pi k}{N}\right)+\frac{ib}{h}\sin\!\left(\frac{2\pi k}{N}\right).
\]
\item Deduzca que la familia es estable en norma 2 uniformemente en $h$ y obtenga una cota explícita para $\|(A_N)^{-1}\|_2$.
\item Compare con el caso del ejercicio anterior: ¿qué papel juegan el término de difusión y el de transporte en la parte real y en la parte imaginaria del símbolo?
\end{enumerate}
\emph{Remark.} La matriz asociada también es circulante, pero la vía pedida en el ejercicio es trabajar directamente con las ecuaciones discretas y los modos de Fourier.
\end{exercise}

\begin{exercise}[Helmholtz periódico]
Considere el problema periódico
\[
  -u''(x)-\omega^2 u(x)=f(x),\qquad x\in(0,1),\qquad u(0)=u(1),\quad u'(0)=u'(1),
\]
con $\omega>0$ fijo.
\begin{enumerate}[(a)]
\item Discretice el problema con diferencias centradas y escriba las ecuaciones del sistema con índices módulo $N$.
\item Muestre, por Fourier discreto, que los modos $w_j^{(k)}=e^{2\pi i kj/N}$ son autofunciones del operador discreto y que los autovalores son
\[
  \lambda_k=\frac{4}{h^2}\sin^2\!\left(\frac{\pi k}{N}\right)-\omega^2,
  \qquad k=0,1,\dots,N-1.
\]
\item Determine para qué valores de $\omega$ y de $N$ el sistema discreto es singular.
\item Discuta, usando únicamente el símbolo de Fourier, cuándo la familia puede ser uniformemente estable en norma 2 y cuándo no.
\end{enumerate}
\emph{Remark.} También aquí la matriz es circulante, pero no hace falta invocar ese hecho: el análisis pedido se obtiene directamente al reemplazar los modos de Fourier en las ecuaciones discretas.
\end{exercise}

\section*{Ecuaciones en derivadas parciales}

En los siguientes ejercicios se trabaja con la ecuación del calor
\[
  u_t = \kappa\,u_{xx},\qquad x\in(0,1),\quad t>0,\quad \kappa>0,
\]
con condición inicial dada y condiciones de borde \textbf{periódicas} $u(0,t)=u(1,t)$, $u_x(0,t)=u_x(1,t)$. Se usan la malla uniforme espacial $x_j=jh$, $h=1/N$, la malla temporal $t^n=n\dt$, el parámetro $\mu:=\kappa\dt/h^2$, y la segunda diferencia centrada
\[
  \delta_{xx}U_j^n := \frac{U_{j-1}^n - 2U_j^n + U_{j+1}^n}{h^2}.
\]

\begin{exercise}[Euler explícito para la ecuación del calor]
\begin{enumerate}[(a)]
\item Aplicando la diferencia hacia adelante en tiempo y la segunda diferencia centrada en espacio, escriba el esquema de diferencias finitas y despeje $U_j^{n+1}$ en función de los valores en el nivel $t^n$. Identifique los coeficientes de $U_{j-1}^n$, $U_j^n$ y $U_{j+1}^n$.

\item \textbf{Error de truncamiento en tiempo.} Fijando $x_j$ y usando la expansión de Taylor en $t$,
\[
  u(x_j,t^{n+1}) = u(x_j,t^n) + \dt\,\partial_t u(x_j,t^n) + \frac{\dt^2}{2}\,\partial_{tt}u(x_j,\tilde{t}),
\]
muestre que el cociente $\bigl[u(x_j,t^{n+1})-u(x_j,t^n)\bigr]/\dt$ aproxima $\partial_t u(x_j,t^n)$ con error $O(\dt)$.

\item \textbf{Error de truncamiento en espacio.} Fijando $t^n$ y usando la expansión de Taylor en $x$,
\[
  u(x_j\pm h, t^n) = u(x_j,t^n) \pm h\,\partial_x u + \frac{h^2}{2}\partial_{xx}u \pm \frac{h^3}{6}\partial_{xxx}u + \frac{h^4}{24}\partial_{xxxx}u(\xi_\pm, t^n),
\]
muestre que $\delta_{xx}u(x_j,t^n) = \partial_{xx}u(x_j,t^n) + O(h^2)$.

\item \textbf{Error de truncamiento global del esquema.} Definiendo
\[
  \tau_j^n := \frac{u(x_j,t^{n+1})-u(x_j,t^n)}{\dt} - \kappa\,\delta_{xx}u(x_j,t^n),
\]
combine los resultados anteriores y el hecho de que $u$ satisface la EDP para mostrar que $\tau_j^n = O(\dt + h^2)$.

\item \textbf{Análisis de estabilidad de Fourier.} Con borde periódico el operador de actualizaci\'on es circulante. Sustituya el modo discreto $V_j^n = G^n e^{i\theta j}$ (con $\theta = 2\pi k/N$, $k=0,\dots,N-1$) en el esquema del inciso~(a) y deduzca el factor de amplificaci\'on
\[
  G_E(\theta) = 1 - 4\mu\sin^2\!\tfrac{\theta}{2}.
\]

\item \textbf{Condición de estabilidad.} La condición de estabilidad de Lax--Richtmyer exige $|G_E(\theta)|\le 1$ para todo $\theta$. Analice para qué valores de $\mu$ se cumple esto y concluya que el esquema es estable si y sólo si
\[
  \mu \le \tfrac{1}{2}.
\]
Interprete esta restricción como una relación entre $\dt$ y $h$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Euler implícito para la ecuación del calor]
\begin{enumerate}[(a)]
\item Aplicando la diferencia hacia adelante en tiempo y la segunda diferencia centrada \textbf{evaluada en el nivel $t^{n+1}$}, escriba el esquema de Euler implícito y reordene para mostrar que cada paso requiere resolver el sistema tridiagonal (periódico)
\[
  -\mu\,U_{j-1}^{n+1} + (1+2\mu)\,U_j^{n+1} - \mu\,U_{j+1}^{n+1} = U_j^n, \qquad j=0,\dots,N-1.
\]

\item \textbf{Error de truncamiento en tiempo.} Desarrolle $u(x_j,t^n)$ alrededor de $t^{n+1}$:
\[
  u(x_j,t^n) = u(x_j,t^{n+1}) - \dt\,\partial_t u(x_j,t^{n+1}) + \frac{\dt^2}{2}\,\partial_{tt}u(x_j,\tilde{t}),
\]
y muestre que $\bigl[u(x_j,t^{n+1})-u(x_j,t^n)\bigr]/\dt = \partial_t u(x_j,t^{n+1}) + O(\dt)$.

\item \textbf{Error de truncamiento global.} Definiendo
\[
  \tau_j^n := \frac{u(x_j,t^{n+1})-u(x_j,t^n)}{\dt} - \kappa\,\delta_{xx}u(x_j,t^{n+1}),
\]
use el inciso anterior y el resultado de la parte~(c) del ejercicio anterior para mostrar que $\tau_j^n = O(\dt + h^2)$.

\item \textbf{Análisis de estabilidad de Fourier.} Sustituya $V_j^n = G^n e^{i\theta j}$ en el esquema del inciso~(a) y deduzca el factor de amplificación
\[
  G_I(\theta) = \frac{1}{1+4\mu\sin^2\!\tfrac{\theta}{2}}.
\]

\item \textbf{Estabilidad incondicional.} Muestre que $0 < G_I(\theta) \le 1$ para todo $\theta$ y todo $\mu>0$, y concluya que Euler implícito es incondicionalmente estable.

\item Compare la restricción de paso en Euler explícito con la ausencia de restricción en Euler implícito para ecuaciones rígidas en espacio (mallas finas).
\end{enumerate}
\end{exercise}

\begin{exercise}[Método de Crank--Nicolson para la ecuación del calor]
El método de Crank--Nicolson promedia las discretizaciones espaciales de Euler explícito e implícito:
\[
  \frac{U_j^{n+1}-U_j^n}{\dt} = \frac{\kappa}{2}\left(\delta_{xx}U_j^n + \delta_{xx}U_j^{n+1}\right).
\]
\begin{enumerate}[(a)]
\item Reordene el esquema para obtener el sistema lineal que debe resolverse en cada paso de tiempo, e identifique la matriz del mismo.

\item \textbf{Error de truncamiento en tiempo.} Use la expansión de la diferencia centrada en tiempo,
\[
  \frac{u(x_j,t^{n+1})-u(x_j,t^n)}{\dt} = \partial_t u\!\left(x_j,t^{n+\frac{1}{2}}\right) + \frac{\dt^2}{24}\,\partial_{ttt}u(x_j,\tilde{t}),
\]
y el promedio espacial
\[
  \frac{1}{2}\kappa\!\left(\delta_{xx}u(x_j,t^n)+\delta_{xx}u(x_j,t^{n+1})\right)
  = \kappa\,\partial_{xx}u\!\left(x_j,t^{n+\frac{1}{2}}\right) + O(\dt^2) + O(h^2),
\]
para mostrar que el error de truncamiento satisface $\tau_j^n = O(\dt^2 + h^2)$, es decir que Crank--Nicolson es de orden 2 en tiempo y en espacio.

\item \textbf{Análisis de estabilidad de Fourier.} Sustituya $V_j^n = G^n e^{i\theta j}$ en el esquema y muestre que el factor de amplificación es
\[
  G_{CN}(\theta) = \frac{1-2\mu\sin^2\!\tfrac{\theta}{2}}{1+2\mu\sin^2\!\tfrac{\theta}{2}}.
\]

\item \textbf{Estabilidad incondicional.} Denotando $s = 2\mu\sin^2(\theta/2) \ge 0$, pruebe que $|(1-s)/(1+s)|\le 1$ para todo $s\ge 0$ y concluya que $|G_{CN}(\theta)|\le 1$ para todo $\mu>0$ y todo $\theta$.

\item Compare el orden de precisión y el costo por paso de Crank--Nicolson con el de Euler explícito e implícito. ¿En qué situaciones conviene usar cada método?
\end{enumerate}
\end{exercise}
