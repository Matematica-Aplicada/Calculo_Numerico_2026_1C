\section*{Métodos de un paso}

\begin{exercise}[Método del punto medio sobre una EDO lineal]
Considere el problema
\[
  y'(t)=-y(t),\qquad y(0)=1,
\]
y el método del punto medio
\[
  y_{n+1}=y_n+h\,f\!\left(t_n+\frac{h}{2},\,y_n+\frac{h}{2}f(t_n,y_n)\right).
\]
\begin{enumerate}[(a)]
\item Escriba la función de transición $\Phi(t,y,h)$ del método.
\item Calcule el error de truncamiento local
\[
  \tau_{n+1}(h)=\frac{y(t_{n+1})-y(t_n)}{h}-\Phi(t_n,y(t_n),h)
\]
y determine su orden.
\item Halle la constante de Lipschitz de $\Phi$ respecto de la variable $y$ y deduzca una cota uniforme para $0<h\le 1$.
\item Use el teorema de convergencia de métodos de un paso para hallar un paso $h$ que garantice
\[
  \max_{t_n\le 10}|y(t_n)-y_n|<10^{-3}.
\]
\end{enumerate}
\end{exercise}

\begin{exercise}[Euler implícito sobre una EDO lineal]
Considere nuevamente
\[
  y'(t)=-y(t),\qquad y(0)=1,
\]
y el método de Euler implícito
\[
  y_{n+1}=y_n-hy_{n+1}.
\]
\begin{enumerate}[(a)]
\item Escriba la función de transición $\Phi(t,y,h)$ del método.
\item Calcule el error de truncamiento local y determine su orden.
\item Calcule la constante de Lipschitz de $\Phi$ respecto de $y$.
\item Use el teorema de convergencia de métodos de un paso para hallar un paso $h$ que garantice
\[
  \max_{t_n\le 10}|y(t_n)-y_n|<10^{-3}.
\]
\item Compare el paso obtenido aquí con el del ejercicio anterior.
\end{enumerate}
\end{exercise}

\begin{exercise}[Método de Heun]
Considere la EDO
\[
  y'(t)=-y(t)+t,\qquad y(0)=1,
\]
y el método de Heun
\begin{align*}
  k_1&=f(t_n,y_n),\\
  k_2&=f(t_n+h,\,y_n+hk_1),\\
  y_{n+1}&=y_n+\frac{h}{2}(k_1+k_2).
\end{align*}
\begin{enumerate}[(a)]
\item Escriba la función de transición $\Phi(t,y,h)$.
\item Halle el orden del error de truncamiento local.
\item Calcule una constante de Lipschitz de $\Phi$ en la variable $y$ para $0<h\le 1$.
\item Use el teorema de convergencia de métodos de un paso para obtener un paso $h$ que garantice error global menor que $10^{-3}$ en el intervalo $[0,10]$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Problema stiff: Euler explícito e implícito]
Considere el problema
\[
  y'=-100\bigl(y-\cos t\bigr)-\sin t,\qquad y(0)=1.
\]
\begin{enumerate}[(a)]
\item Verifique que la solución exacta es $y(t)=\cos t$.
\item Escriba las funciones de transición $\Phi_E$ y $\Phi_I$ de Euler explícito e implícito.
\item Calcule las constantes de Lipschitz de $\Phi_E$ y $\Phi_I$ respecto de $y$.
\item Explique, usando la cota del teorema de convergencia de un paso, por qué el método explícito se deteriora cuando la constante de Lipschitz es grande.
\item Compare este análisis con el comportamiento cualitativo de ambos métodos sobre la ecuación test $y'=\lambda y$ con $\lambda<0$.
\end{enumerate}
\end{exercise}


\section*{Problemas de valores de contorno}

\begin{exercise}[Transporte-reacción periódico]
Considere el problema periódico
\[
  b\,u'(x)+\sigma u(x)=f(x),\qquad x\in(0,1),\qquad u(0)=u(1),\quad u'(0)=u'(1),
\]
con $b\in\R$ y $\sigma>0$. En la malla uniforme $x_j=jh$, $h=1/N$, use diferencia centrada para $u'$:
\[
  b\,\frac{u_{j+1}-u_{j-1}}{2h}+\sigma u_j=f(x_j),\qquad j=0,\dots,N-1,
\]
con índices módulo $N$.
\begin{enumerate}[(a)]
\item Verifique que los modos de Fourier discretos $w_j^{(k)}=e^{2\pi i kj/N}$ satisfacen las condiciones de borde periódicas.
\item Muestre, por sustitución directa de $w^{(k)}$ en las ecuaciones del sistema discreto, que estos modos son autofunciones del operador de diferencias finitas y que los autovalores son
\[
  \lambda_k=\sigma+\frac{ib}{h}\sin\!\left(\frac{2\pi k}{N}\right).
\]
\item Deduzca que $|\lambda_k|\ge \sigma$ para todo $k$ y concluya que $\|(A_N)^{-1}\|_2\le 1/\sigma$.
\item Muestre que si $f\in C^2([0,1])$, el error de truncamiento es $O(h^2)$ y que por ende la convergencia también es $O(h^2)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Advección-difusión-reacción periódica]
Considere
\[
  -\nu u''(x)+b\,u'(x)+\sigma u(x)=f(x),\qquad x\in(0,1),
\]
con condiciones periódicas, donde $\nu>0$, $b\in\R$ y $\sigma>0$.
\begin{enumerate}[(a)]
\item Discretice $u''$ y $u'$ con diferencias centradas y escriba las ecuaciones del esquema con índices módulo $N$.
\item Muestre que si $f\in C^2([0,1])$ el error de truncamiento es $O(h^2)$.
\item Usando Fourier discreto, pruebe que los autovalores son
\[
  \lambda_k=\sigma+\frac{4\nu}{h^2}\sin^2\!\left(\frac{\pi k}{N}\right)+\frac{ib}{h}\sin\!\left(\frac{2\pi k}{N}\right).
\]
\item Deduzca que la familia es estable en norma 2 uniformemente en $h$ y obtenga una cota explícita para $\|(A_N)^{-1}\|_2$. Concluya que la convergencia es $O(h^2)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Helmholtz periódico]
Considere el problema periódico
\[
  -u''(x)-\omega^2 u(x)=f(x),\qquad x\in(0,1),\qquad u(0)=u(1),\quad u'(0)=u'(1),
\]
con $\omega>0$ fijo.
\begin{enumerate}[(a)]
\item Discretice el problema con diferencias centradas y escriba las ecuaciones del sistema con índices módulo $N$.
\item Muestre, por Fourier discreto, que los modos $w_j^{(k)}=e^{2\pi i kj/N}$ son autofunciones del operador discreto y que los autovalores son
\[
  \lambda_k=\frac{4}{h^2}\sin^2\!\left(\frac{\pi k}{N}\right)-\omega^2,
  \qquad k=0,1,\dots,N-1.
\]
\item Determine para qué valores de $\omega$ y de $N$ el sistema discreto es singular.
\item Discuta, usando únicamente el símbolo de Fourier, cuándo la familia puede ser uniformemente estable en norma 2 y cuándo no.
\end{enumerate}
\end{exercise}

\section*{Ecuaciones en derivadas parciales}

En todos los ejercicios se trabaja con condiciones de borde \textbf{periódicas} y malla uniforme $x_j=jh$, $h=1/N$, $t^n=n\dt$. Se usa la segunda diferencia centrada
\[
  \delta_{xx}U_j^n := \frac{U_{j-1}^n - 2U_j^n + U_{j+1}^n}{h^2}
\]
y la diferencia centrada de primer orden
\[
  \delta_x U_j^n := \frac{U_{j+1}^n - U_{j-1}^n}{2h},
\]
con índices módulo $N$.

\begin{exercise}[Método de Crank--Nicolson para la ecuación del calor]
Consideramos la ecuación del calor
\[
  u_t = \kappa\,u_{xx},\qquad \kappa>0,
\]
con condiciones de borde periódicas y parámetro $\mu:=\kappa\dt/h^2$.
El método de Crank--Nicolson promedia las discretizaciones espaciales explícita e implícita:
\[
  \frac{U_j^{n+1}-U_j^n}{\dt} = \frac{\kappa}{2}\left(\delta_{xx}U_j^n + \delta_{xx}U_j^{n+1}\right).
\]
\begin{enumerate}[(a)]
\item Reordene el esquema para obtener el sistema lineal que debe resolverse en cada paso de tiempo e identifique la matriz del mismo.

\item \textbf{Error de truncamiento.} Use la expansión
\[
  \frac{u(x_j,t^{n+1})-u(x_j,t^n)}{\dt} = \partial_t u\!\left(x_j,t^{n+\frac{1}{2}}\right) + O(\dt^2)
\]
y el promedio espacial
\[
  \tfrac{1}{2}\kappa\!\left(\delta_{xx}u(x_j,t^n)+\delta_{xx}u(x_j,t^{n+1})\right)
  = \kappa\,\partial_{xx}u\!\left(x_j,t^{n+\frac{1}{2}}\right) + O(\dt^2) + O(h^2),
\]
para concluir que $\tau_j^n = O(\dt^2 + h^2)$.

\item \textbf{Análisis de estabilidad de Fourier.} Sustituya el modo $V_j^n = G^n e^{i\theta j}$ en el esquema y deduzca el factor de amplificación
\[
  G_{CN}(\theta) = \frac{1-2\mu\sin^2\!\tfrac{\theta}{2}}{1+2\mu\sin^2\!\tfrac{\theta}{2}}.
\]

\item \textbf{Estabilidad incondicional.} Denotando $s = 2\mu\sin^2(\theta/2) \ge 0$, pruebe que $|{(1-s)}/{(1+s)}|\le 1$ para todo $s\ge 0$.

\item Compare el orden de precisión en $\dt$ con el de Euler explícito e implícito. ¿Qué ventaja tiene Crank--Nicolson respecto del costo adicional?
\end{enumerate}
\end{exercise}

\begin{exercise}[Ecuación de difusión-reacción, Euler explícito]
Considere la ecuación de difusión-reacción
\[
  u_t = \kappa\,u_{xx} - \sigma u,\qquad \kappa>0,\ \sigma>0,
\]
con condiciones de borde periódicas. Se denota $\mu:=\kappa\dt/h^2$.
\begin{enumerate}[(a)]
\item Aplique diferencia centrada en espacio y Euler explícito en tiempo. Despeje $U_j^{n+1}$ y muestre que el esquema tiene la forma
\[
  U_j^{n+1} = \mu\,U_{j-1}^n + (1-2\mu-\sigma\dt)\,U_j^n + \mu\,U_{j+1}^n.
\]

\item \textbf{Error de truncamiento.} Muestre que $\tau_j^n = O(\dt + h^2)$.

\item \textbf{Factor de amplificación.} Sustituya $V_j^n = G^n e^{i\theta j}$ en el esquema y deduzca
\[
  G_E(\theta) = 1 - 4\mu\sin^2\!\tfrac{\theta}{2} - \sigma\dt.
\]
Observe que $G_E$ es real.

\item \textbf{Condición de estabilidad fuerte.} La condición de estabilidad fuerte $|G_E(\theta)|\le 1$ para todo $\theta$ se descompone en cota superior e inferior. Muestre que la cota superior se satisface automáticamente, y que la cota inferior da
\[
  4\mu + \sigma\dt \le 2,\qquad\text{es decir,}\qquad
  \mu \le \frac{1-\sigma\dt/2}{2}.
\]

\item \textbf{Interpretación.} Suponga que $\sigma$ es grande (reacción rígida). ¿Qué restricción impone sobre $\dt$ independientemente de $h$? Compare con el caso $\sigma=0$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Ecuación de difusión-reacción, Euler implícito]
Considere la ecuación de difusión-reacción
\[
  u_t = \kappa\,u_{xx} - \sigma u,\qquad \kappa>0,\ \sigma>0,
\]
con condiciones de borde periódicas. Se denota $\mu:=\kappa\dt/h^2$.
\begin{enumerate}[(a)]
\item Aplique diferencia centrada en espacio y Euler implícito en tiempo. Reordene y muestre que el sistema a resolver en cada paso es
\[
  -\mu\,U_{j-1}^{n+1} + (1+2\mu+\sigma\dt)\,U_j^{n+1} - \mu\,U_{j+1}^{n+1} = U_j^n.
\]

\item \textbf{Error de truncamiento.} Muestre que $\tau_j^n = O(\dt + h^2)$.

\item \textbf{Factor de amplificación.} Sustituya $V_j^n = G^n e^{i\theta j}$ y deduzca
\[
  G_I(\theta) = \frac{1}{1 + 4\mu\sin^2\!\tfrac{\theta}{2} + \sigma\dt}.
\]

\item \textbf{Estabilidad incondicional.} Use que $\mu\ge 0$, $\sigma\dt\ge 0$ y $\sin^2(\theta/2)\ge 0$ para concluir que
\[
  0 < G_I(\theta) \le 1\qquad\text{para todo }\theta,
\]
y por tanto el esquema es incondicionalmente estable.

\item \textbf{Comparación con Euler explícito.} Compare este resultado con la condición
\[
  4\mu + \sigma\dt \le 2
\]
del esquema explícito. Discuta qué ocurre cuando $\sigma$ es grande (reacción rígida).
\end{enumerate}
\end{exercise}

\begin{exercise}[Ecuación de convección-difusión, Euler implícito]
Considere la ecuación de convección-difusión
\[
  u_t + c\,u_x = \varepsilon\,u_{xx},\qquad c\in\mathbb{R},\ \varepsilon>0,
\]
con condiciones de borde periódicas. Se denota $\mu_\varepsilon:=\varepsilon\dt/h^2$ (número difusivo) y $\nu:=c\dt/h$ (número de Courant).
\begin{enumerate}[(a)]
\item Use Euler implícito en tiempo, diferencia centrada de primer orden para $u_x$ y segunda diferencia centrada para $u_{xx}$, todos evaluados en el nivel $t^{n+1}$. Reordene y muestre que el sistema en cada paso tiene la forma
\[
  \left(-\mu_\varepsilon - \tfrac{\nu}{2}\right)U_{j-1}^{n+1}
  + (1+2\mu_\varepsilon)\,U_j^{n+1}
  + \left(-\mu_\varepsilon + \tfrac{\nu}{2}\right)U_{j+1}^{n+1}
  = U_j^n.
\]

\item \textbf{Error de truncamiento.} Argumente que el esquema es de orden $O(\dt + h^2)$ en el error de truncamiento.

\item \textbf{Factor de amplificación.} Sustituya $V_j^n = G^n e^{i\theta j}$ y use $e^{i\theta}+e^{-i\theta}=2\cos\theta$, $e^{i\theta}-e^{-i\theta}=2i\sin\theta$ para mostrar que
\[
  G_I(\theta) = \frac{1}{1 + 4\mu_\varepsilon\sin^2\!\tfrac{\theta}{2} + i\nu\sin\theta}.
\]

\item \textbf{Estabilidad incondicional.} Calcule $|G_I(\theta)|^2$ y deduzca que $|G_I(\theta)|\le 1$ para todo $\mu_\varepsilon>0$, todo $\nu$ y todo $\theta$. Concluya que el esquema es incondicionalmente estable.

\item Discuta qué rol juegan los términos de difusión y de convección en la parte real e imaginaria del denominador de $G_I$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Advección pura, Euler explícito centrado]
Considere la ecuación de advección pura
\[
  u_t + c\,u_x = 0,\qquad c\ne 0,
\]
con condiciones de borde periódicas. Se denota $\nu:=c\dt/h$.
\begin{enumerate}[(a)]
\item Use Euler explícito en tiempo y diferencia centrada de primer orden en espacio para $u_x$. Despeje $U_j^{n+1}$ y escriba el esquema.

\item \textbf{Error de truncamiento.} Muestre que $\tau_j^n = O(\dt + h^2)$.

\item \textbf{Factor de amplificación.} Sustituya $V_j^n = G^n e^{i\theta j}$ y deduzca
\[
  G(\theta) = 1 - i\nu\sin\theta.
\]

\item \textbf{Inestabilidad incondicional.} Calcule $|G(\theta)|^2 = 1 + \nu^2\sin^2\theta$ y concluya que $|G(\theta)|>1$ para todo $\nu\ne 0$ y todo $\theta\ne 0,\pi$. Deduzca que el esquema es inestable para cualquier elección de $\dt$ y $h$.

\item Contraste con el resultado del ejercicio anterior: ¿qué papel cumple el término de difusión $\varepsilon u_{xx}$ para recuperar la estabilidad?
\end{enumerate}
\end{exercise}
