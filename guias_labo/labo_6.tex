\bej
\textbf{(Intersección de curvas planas: método de Newton en $\R^2$.)}

Considere el sistema de ecuaciones
\begin{equation}\label{eq:sistema}
  F(x,y) =
  \begin{pmatrix} f_1(x,y) \\ f_2(x,y) \end{pmatrix}
  =
  \begin{pmatrix} x^2 + y^2 + xy - 7 \\ x^3 - y + 1 \end{pmatrix}
  = \mathbf{0}.
\end{equation}
Geométricamente, $f_1 = 0$ es una elipse (rotada) y $f_2 = 0$ es la curva cúbica $y = x^3 + 1$.

\begin{enumerate}[(a)]
  \item Verifique algebraicamente que $(x_\ast, y_\ast) = (1, 2)$ es solución exacta de~\eqref{eq:sistema}.

  \item Calcule analíticamente la matriz jacobiana
    \[
      J(x,y) = \frac{\partial F}{\partial(x,y)}
             = \begin{pmatrix}
                 \partial_x f_1 & \partial_y f_1 \\
                 \partial_x f_2 & \partial_y f_2
               \end{pmatrix}.
    \]
    Evalúe $J(1,2)$ y verifique que es invertible.

  \item El método de Newton para $F(x,y)=\mathbf{0}$ es la iteración
    \[
      \mathbf{p}^{(k+1)} = \mathbf{p}^{(k)} - J\!\left(\mathbf{p}^{(k)}\right)^{-1} F\!\left(\mathbf{p}^{(k)}\right).
    \]
    Implemente esta iteración. En la práctica, en lugar de invertir $J$, resuelva el sistema lineal $J \delta = -F$ y actualice $\mathbf{p} \leftarrow \mathbf{p} + \delta$ (use \texttt{numpy.linalg.solve}).
    Itere hasta que $\|F(\mathbf{p}^{(k)})\| < 10^{-10}$.

  \item Ejecute Newton a partir de $\mathbf{p}^{(0)} = (0, 0)$. Registre $\|F(\mathbf{p}^{(k)})\|$ en cada iteración y grafíquelo en escala semilogarítmica. ¿En cuántas iteraciones converge? ¿Observa convergencia cuadrática?

  \item La curva cúbica y la elipse se intersectan en más de un punto. Grafique ambas curvas
    en la región $[-3, 3] \times [-3, 6]$ e identifique visualmente una segunda solución.
    Ejecute Newton desde un punto inicial próximo a ella y repórtela.

\end{enumerate}
\fej

\bej
\textbf{(Descenso por gradiente para regresión logística: dataset \textit{Titanic}.)}

Se desea predecir la supervivencia de un pasajero del Titanic a partir de tres atributos:
género (\texttt{sex}), clase del pasaje (\texttt{pclass}) y edad (\texttt{age}).
El modelo es una \textit{regresión logística}: dado el vector de atributos extendido
$\mathbf{x} \in \R^4$ (los tres atributos más un $1$ para el sesgo), la probabilidad
predicha de supervivencia es
\[
  \hat{p}(\mathbf{x};\, \mathbf{w}) = \sigma(\mathbf{w}^\top \mathbf{x}),
  \qquad \sigma(z) = \frac{1}{1+e^{-z}}.
\]
El parámetro $\mathbf{w} \in \R^4$ se ajusta minimizando la
\textit{pérdida cuadrática}
\[
  L(\mathbf{w}) = \frac{1}{n} \sum_{i=1}^{n} \bigl(\hat{p}(\mathbf{x}_i;\,\mathbf{w}) - y_i\bigr)^2,
\]
donde $y_i \in \{0,1\}$ es la etiqueta de supervivencia del pasajero $i$.

\begin{enumerate}[(a)]
  \item \textbf{Preparación de datos.} Cargue el dataset con
    \begin{center}
      \texttt{import seaborn as sns; df = sns.load\_dataset('titanic')}
    \end{center}
    Construya la matriz $X \in \R^{n \times 4}$ y el vector $\mathbf{y} \in \{0,1\}^n$ de la siguiente forma:
    \begin{itemize}
      \item Elimine las filas con valores faltantes en \texttt{age}, \texttt{sex} o \texttt{pclass}.
      \item Codifique \texttt{sex}: \texttt{male} $\mapsto 0$, \texttt{female} $\mapsto 1$.
      \item Normalice \texttt{age} y \texttt{pclass} restando la media y dividiendo por el desvío estándar.
      \item Añada una columna de unos (sesgo) como última columna de $X$.
      \item Tome $\mathbf{y}$ de la columna \texttt{survived}.
    \end{itemize}
    Reporte $n$, la fracción de supervivientes y las dimensiones de $X$.

  \item \textbf{Gradiente de la pérdida.} Muestre que
    \[
      \nabla_{\mathbf{w}} L(\mathbf{w})
        = \frac{2}{n} \sum_{i=1}^{n}
          \bigl(\hat{p}_i - y_i\bigr)\, \hat{p}_i(1 - \hat{p}_i)\, \mathbf{x}_i,
    \]
    donde $\hat{p}_i = \sigma(\mathbf{w}^\top \mathbf{x}_i)$.
    En forma matricial, si $\hat{\mathbf{p}} = \sigma(X\mathbf{w})$ (componente a componente):
    \[
      \nabla_{\mathbf{w}} L(\mathbf{w})
        = \frac{2}{n}\, X^\top \bigl[(\hat{\mathbf{p}} - \mathbf{y}) \odot
          \hat{\mathbf{p}} \odot (1 - \hat{\mathbf{p}})\bigr],
    \]
    donde $\odot$ denota el producto de Hadamard (componente a componente).
    Verifique numéricamente esta fórmula comparándola con diferencias finitas centradas
    para un $\mathbf{w}$ de prueba.

  \item \textbf{Implementación.} Programe la función
    \texttt{descenso\_gradiente(X, y, alpha, n\_iter)} que, partiendo de $\mathbf{w}^{(0)} = \mathbf{0}$,
    ejecute $n_{\mathrm{iter}}$ pasos de descenso con paso constante $\alpha$:
    \[
      \mathbf{w}^{(k+1)} = \mathbf{w}^{(k)} - \alpha \,\nabla L\!\left(\mathbf{w}^{(k)}\right).
    \]
    La función debe devolver el historial de pérdidas $\{L(\mathbf{w}^{(k)})\}_{k=0}^{n_{\mathrm{iter}}}$.

  \item \textbf{Efecto del paso $\alpha$.} Ejecute el método para
    $\alpha \in \{0.01,\, 0.1,\, 1.0,\, 5.0\}$ con $n_{\mathrm{iter}} = 500$.
    Grafique en una misma figura las curvas de pérdida vs.\ iteración.
    ¿Qué ocurre con $\alpha$ demasiado grande? ¿Converge para todos los valores?

  \item \textbf{Evaluación.} Con el $\alpha$ que produjo mejor convergencia,
    calcule la \textit{accuracy} (fracción de predicciones correctas) usando el criterio
    $\hat{y}_i = \mathbf{1}[\hat{p}_i \geq 0.5]$.
    Compare este resultado con el clasificador trivial que siempre predice la clase mayoritaria.

\end{enumerate}
\fej
